323        #Private function to check and recursively fill in pixels.
324        def fill_pixel(self,current_position,canvas_obj,color):
325
326            if current_position[0] >= 0 and \
327                current_position[0] <= canvas_obj.width-1 and \
328                current_position[1] >= 0 and \
329                current_position[1] <= canvas_obj.height-1:
330            
331                old_color = canvas_obj.surface.get_at(current_position)
332                if old_color == color:
333                    return
334
335                stack = [current_position]
336                
337                while stack:
338
339                    x, y = stack.pop()
340
341                    if canvas_obj.surface.get_at((x, y)) == old_color:
342                        canvas_obj.surface.set_at((x, y), color)
343
344                        #Add left pixel if valid index
345                        if x > 0:
346                            stack.append((x-1, y))
347
348                        #Add right pixel if valid index
349                        if x < canvas_obj.width - 1:
350                            stack.append((x+1, y))
351
352                        #Add bottom pixel if valid index
353                        if y > 0:
354                            stack.append((x, y-1))
355
356                        #Add top pixel if valid index
357                        if y < canvas_obj.height - 1:
358                            stack.append((x, y+1))

324        #Private function to check and recursively fill in pixels.
325        def fill_pixel(self,current_position,canvas_obj,color):
326
327            if current_position[0] >= 0 and \
328                current_position[0] <= canvas_obj.width-1 and \
329                current_position[1] >= 0 and \
330                current_position[1] <= canvas_obj.height-1:
331            
332                old_color = canvas_obj.surface.get_at(current_position)
333                if old_color == color:
334                    return
335
336                stack = [current_position]
337                
338                while stack:
339
340                    x, y = stack.pop()
341
342                    if canvas_obj.surface.get_at((x, y)) == old_color:
343                        canvas_obj.surface.set_at((x, y), color)
344
345                        #Add left pixel if valid index
346                        if x > 0:
347                            stack.append((x-1, y))
348
349                        #Add right pixel if valid index
350                        if x < canvas_obj.width - 1:
351                            stack.append((x+1, y))
352
353                        #Add bottom pixel if valid index
354                        if y > 0:
355                            stack.append((x, y-1))
356
357                        #Add top pixel if valid index
358                        if y < canvas_obj.height - 1:
359                            stack.append((x, y+1))
